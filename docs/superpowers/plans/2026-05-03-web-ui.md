# Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development or superpowers:executing-plans.

**Goal:** Build the Next.js dashboard: login + protected routes, site dashboard with live cards, control panel with state-machine UX, device detail with metric grid, admin pages for catalog/profiles. Generates a typed TypeScript SDK from the cloud's OpenAPI; uses NextAuth (Auth.js v5) Credentials provider; multiplexes one WebSocket per device across all subscribers.

**Architecture:** Next.js App Router + RSC. Server components fetch initial data via `apiServer()` (built on `openapi-fetch` with auth interceptor). Client components subscribe to `useDeviceLiveStream` (singleton-pool hook), backed by TanStack Query for cached state. One WS connection per device shared across all dashboard cards. Auth token lives in NextAuth's encrypted JWT cookie; server-side session derived via `auth()`.

**Tech Stack:** Next.js 15, React 19, TypeScript 5.5, Tailwind 4, shadcn/ui, Tremor 3, Auth.js v5 (NextAuth), TanStack Query 5, openapi-fetch + openapi-typescript, Vitest, Playwright.

```typescript
// packages/web_ui/lib/auth-helpers.ts
export function isProtectedPath(pathname: string): boolean {
  if (pathname === '/login' || pathname.startsWith('/login/')) return false;
  if (pathname.startsWith('/api/auth/')) return false;
  if (pathname.startsWith('/_next/')) return false;
  return true;
}
```

```typescript
// packages/web_ui/auth.ts
import NextAuth from 'next-auth';
import Credentials from 'next-auth/providers/credentials';

const API_BASE = process.env.NEXT_PUBLIC_API_BASE ?? '';

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [
    Credentials({
      credentials: {
        email: { label: 'Email', type: 'email' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(creds) {
        const res = await fetch(`${API_BASE}/api/v1/auth/login`, {
          method: 'POST',
          headers: { 'content-type': 'application/json' },
          body: JSON.stringify(creds),
        });
        if (!res.ok) return null;
        const { access_token, expires_in, user } = await res.json();
        return {
          id: user.id,
          email: user.email,
          name: user.display_name,
          accessToken: access_token,
          tier: user.tier,
          role: user.role,
          expiresAt: Date.now() + expires_in * 1000,
        } as any;
      },
    }),
  ],
  pages: { signIn: '/login' },
  session: { strategy: 'jwt', maxAge: 24 * 60 * 60 },
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        const u = user as any;
        token.accessToken = u.accessToken;
        token.tier = u.tier;
        token.role = u.role;
        token.expiresAt = u.expiresAt;
      }
      return token;
    },
    async session({ session, token }) {
      (session as any).accessToken = token.accessToken;
      (session as any).expiresAt = token.expiresAt;
      if (session.user) {
        (session.user as any).tier = token.tier;
        (session.user as any).role = token.role;
      }
      return session;
    },
  },
});
```

```typescript
import { auth } from '@/auth';
import { NextResponse } from 'next/server';

import { isProtectedPath } from '@/lib/auth-helpers';

export default auth((req) => {
  const protectedRoute = isProtectedPath(req.nextUrl.pathname);
  if (protectedRoute && !req.auth) {
    const cb = encodeURIComponent(req.nextUrl.pathname + req.nextUrl.search);
    return NextResponse.redirect(new URL(`/login?callbackUrl=${cb}`, req.url));
  }
  if (!protectedRoute && req.auth && req.nextUrl.pathname === '/login') {
    return NextResponse.redirect(new URL('/', req.url));
  }
});
```

```tsx
import { Label } from '@/components/ui/label';

const Schema = z.object({ email: z.string().email(), password: z.string().min(1) });
type Form = z.infer<typeof Schema>;

export default function LoginPage() {
  const router = useRouter();
  const params = useSearchParams();
  const callbackUrl = params.get('callbackUrl') ?? '/';
  const [error, setError] = useState<string | null>(null);
  const { register, handleSubmit, formState: { isSubmitting } } = useForm<Form>({
    resolver: zodResolver(Schema),

    if (result?.error) {
      setError('Email or password incorrect');
      return;
    }
    router.push(callbackUrl);
    router.refresh();
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4 w-full max-w-sm bg-white p-8 rounded shadow">
      <h1 className="text-xl font-semibold">Sign in</h1>

        <Input id="email" type="email" autoComplete="username" {...register('email')} />
      </div>
      <div>
        <Label htmlFor="password">Password</Label>

      </div>
      {error && <p className="text-sm text-red-600">{error}</p>}
      <Button type="submit" disabled={isSubmitting} className="w-full">Sign in</Button>
```

```typescript
// packages/web_ui/lib/api-client.ts
'use client';

import createClient, { type Client } from 'openapi-fetch';
import { auth } from '@/auth';
import { redirect } from 'next/navigation';

import type { paths } from './api-schema';

export class SessionExpiredError extends Error {}

export function buildClient(accessToken: string, baseUrl: string): Client<paths> {
  const client = createClient<paths>({ baseUrl: `${baseUrl}/api/v1` });
  client.use({
    onRequest({ request }) {
      request.headers.set('authorization', `Bearer ${accessToken}`);
      return request;
    },
    onResponse({ response }) {
      if (response.status === 401) throw new SessionExpiredError();
      return response;
    },
  });
  return client;
}

const API_BASE = process.env.NEXT_PUBLIC_API_BASE ?? '';

export async function apiServer(): Promise<Client<paths>> {
  const session = await auth();
  const token = (session as any)?.accessToken;
  if (!token) redirect('/login');
  return buildClient(token, API_BASE);
}
```

```typescript
// packages/web_ui/lib/ws-client.ts
'use client';

import { useEffect, useRef } from 'react';
import { useSession } from 'next-auth/react';

type ReadingMessage = { type: 'reading'; data: { time: string; metric: string; value: number; quality: string } };
type SnapshotMessage = { type: 'snapshot'; data: { snapshot_time: string; metrics: Record<string, unknown> } };
type HeartbeatMessage = { type: 'heartbeat'; data: Record<string, unknown> };
type LiveMessage = ReadingMessage | SnapshotMessage | HeartbeatMessage;

type Subscriber = {
  onReading?: (msg: ReadingMessage) => void;
  onSnapshot?: (msg: SnapshotMessage) => void;
  onHeartbeat?: (msg: HeartbeatMessage) => void;
};

type PoolEntry = {
  ws: WebSocket;
  subscribers: Set<Subscriber>;
  reconnectScheduled: boolean;
  reconnectAttempts: number;
};

const pool = new Map<string, PoolEntry>();
const WS_BASE = process.env.NEXT_PUBLIC_WS_BASE ?? '';

const BACKOFF_SCHEDULE = [1_000, 2_000, 4_000, 8_000, 16_000, 30_000];

export function _resetPoolForTests() { pool.clear(); }
export function _peekReconnectScheduled(deviceId: string) {
  return pool.get(deviceId)?.reconnectScheduled ?? false;
}

function openConnection(deviceId: string, token: string, slug: string): PoolEntry {
  const url = `${WS_BASE}/sites/${slug}/devices/${deviceId}/live`;
  const ws = new WebSocket(url, ['solamon-bearer', token]);
  const entry: PoolEntry = { ws, subscribers: new Set(), reconnectScheduled: false, reconnectAttempts: 0 };

  ws.onmessage = (event: { data: string }) => {
    let msg: LiveMessage;
    try { msg = JSON.parse(event.data); } catch { return; }
    for (const sub of entry.subscribers) {
      switch (msg.type) {
        case 'reading':   sub.onReading?.(msg); break;
        case 'snapshot':  sub.onSnapshot?.(msg); break;
        case 'heartbeat': sub.onHeartbeat?.(msg); break;
      }
    }
  };

  ws.onclose = (event: { code: number }) => {
    if (event.code === 1000 || event.code === 1008 || entry.subscribers.size === 0) {
      pool.delete(deviceId);
      return;
    }
    entry.reconnectScheduled = true;
    const delay = BACKOFF_SCHEDULE[Math.min(entry.reconnectAttempts, BACKOFF_SCHEDULE.length - 1)];
    setTimeout(() => {
      const fresh = openConnection(deviceId, token, slug);
      fresh.subscribers = entry.subscribers;
      fresh.reconnectAttempts = entry.reconnectAttempts + 1;
      pool.set(deviceId, fresh);
    }, delay);
  };

  pool.set(deviceId, entry);
  return entry;
}

export function acquire(deviceId: string, token: string, subscriber: Subscriber, slug: string = 'bench'): () => void {
  let entry = pool.get(deviceId);
  if (!entry) entry = openConnection(deviceId, token, slug);
  entry.subscribers.add(subscriber);
  return () => {
    const e = pool.get(deviceId);
    if (!e) return;
    e.subscribers.delete(subscriber);
    if (e.subscribers.size === 0) {
      e.ws.close();
      pool.delete(deviceId);
    }
  };
}

export function useDeviceLiveStream(deviceId: string, slug: string, handlers: Subscriber): void {
  const session = useSession();
  const handlersRef = useRef(handlers);
  handlersRef.current = handlers;

  useEffect(() => {
    const token = (session.data as any)?.accessToken;
    if (!token) return;
    const subscriber: Subscriber = {
      onReading: (m) => handlersRef.current.onReading?.(m),
      onSnapshot: (m) => handlersRef.current.onSnapshot?.(m),
      onHeartbeat: (m) => handlersRef.current.onHeartbeat?.(m),
    };
    return acquire(deviceId, token, subscriber, slug);
  }, [deviceId, slug, (session.data as any)?.accessToken]);
}
```

```tsx
import { createContext, useContext, useEffect, useState } from 'react';

const Ctx = createContext<number>(0);

export function RelativeTimeTickProvider({ children }: { children: React.ReactNode }) {
  const [tick, setTick] = useState(0);
```

```tsx
import { useQuery, useQueryClient } from '@tanstack/react-query';
import { Card } from '@tremor/react';

import { useApiClient } from '@/lib/use-api-client';
import { useDeviceLiveStream } from '@/lib/ws-client';

type Snapshot = { snapshot_time: string; metrics: Record<string, number | string | null> };

export function PowerHeroTile({ slug, deviceId, initial }: { slug: string; deviceId: string; initial: Snapshot }) {
  const client = useApiClient();
  const qc = useQueryClient();

  const { data } = useQuery<Snapshot>({
    queryKey: ['snapshot', deviceId],
    queryFn: async () => {
      const { data, error } = await client!.GET(
        '/sites/{slug}/devices/{device_id}/snapshot' as any,
        { params: { path: { slug, device_id: deviceId } } } as any,
      );
      if (error) throw error;
      return data as Snapshot;
    },
    initialData: initial,
    enabled: !!client,
    staleTime: 30_000,
  });

  useDeviceLiveStream(deviceId, slug, {
    onSnapshot: (msg) => {
      qc.setQueryData(['snapshot', deviceId], msg.data);
    },
    onReading: (msg) => {
      if (msg.data.metric === 'active_power_total') {
        qc.setQueryData<Snapshot>(['snapshot', deviceId], (prev) => prev && {
          ...prev,
          metrics: { ...prev.metrics, active_power_total: msg.data.value },
          snapshot_time: msg.data.time,
        });
      }
    },
  });

  const value = (data?.metrics?.active_power_total as number | null | undefined) ?? null;
  return (
    <Card className="p-4">
      <div className="text-sm text-slate-500">Active power</div>
      <div className="text-3xl font-semibold tabular-nums mt-1">
        {value === null ? '—' : `${value.toFixed(2)} kW`}
      </div>
    </Card>
  );
}
```
