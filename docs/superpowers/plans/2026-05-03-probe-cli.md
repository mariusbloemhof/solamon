# Probe CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development or superpowers:executing-plans.

**Goal:** Standalone CLI a commissioning operator runs against a freshly-cabled device. Connects via Modbus TCP, runs every committed profile's `fingerprint_match`, ranks results by confidence, and emits a JSON report conforming to `architecture/probe-output.schema.json`. Used during Day 1 of bench rehearsal and during every site install before kicking off the agent.

**Architecture:** Pure Python CLI. One Modbus connection (pymodbus). For each profile in `architecture/profiles/`, run `Profile.fingerprint_match()` and collect results. Rank by confidence; if `--reference-value` flag is supplied, do a value-cross-check against the highest-ranked profile. Output is JSON (machine-readable) + a human-readable summary.

**Tech Stack:** Python 3.12, pymodbus 3.x, typer, rich (for human-readable output), pytest, pytest-asyncio.

**Linear:** SOL-18.

**Conventions:** [`2026-05-03-conventions.md`](2026-05-03-conventions.md).

```python
# packages/probe_cli/src/probe_cli/runner.py
"""Run fingerprint_match against every committed profile; rank results."""
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

from profile_loader import Catalog, FingerprintResult, Profile, ProfileLoader


CONFIDENCE_RANK = {"positive": 0, "negative_fingerprint": 1, "none": 2}


@dataclass(frozen=True)
class ProfileMatch:
    profile_slug: str
    manufacturer: str
    model: str
    match: bool
    confidence: str
    identifiers: dict[str, Any] = field(default_factory=dict)
    failures: list[str] = field(default_factory=list)


@dataclass(frozen=True)
class ProbeResult:
    timestamp: str
    target: dict[str, Any]                       # host, port, unit_id
    matches: list[ProfileMatch]                  # all candidates, sorted by confidence


def _rank_key(m: ProfileMatch) -> tuple[int, int]:
    # Matches before non-matches; positive before negative_fingerprint.
    return (0 if m.match else 1, CONFIDENCE_RANK[m.confidence])


async def run_probe(
    *, client: Any, profiles_dir: Path, catalog_path: Path,
    unit_id: int, host: str = "", port: int = 502,
) -> ProbeResult:
    from datetime import datetime, timezone
    loader = ProfileLoader()
    catalog = loader.load_catalog(catalog_path)
    matches: list[ProfileMatch] = []
    for profile_path in sorted(profiles_dir.glob("*.yaml")):
        try:
            profile = loader.load_profile(profile_path, catalog)
        except Exception:
            continue
        result: FingerprintResult = await profile.fingerprint_match(client, unit_id=unit_id)
        matches.append(ProfileMatch(
            profile_slug=profile_path.stem,
            manufacturer=profile.device.manufacturer,
            model=profile.device.model,
            match=result.match,
            confidence=result.confidence,
            identifiers=dict(result.identifiers),
            failures=list(result.failures),
        ))
    matches.sort(key=_rank_key)
    return ProbeResult(
        timestamp=datetime.now(timezone.utc).isoformat(timespec="milliseconds").replace("+00:00", "Z"),
        target={"host": host, "port": port, "unit_id": unit_id},
        matches=matches,
    )
```

```python
# packages/probe_cli/src/probe_cli/output.py
"""Probe output: JSON (schema-conformant) + rich human-readable rendering."""
from __future__ import annotations

import json
from dataclasses import asdict
from typing import Any

from rich.console import Console
from rich.table import Table

from .runner import ProbeResult


def to_json(result: ProbeResult) -> dict[str, Any]:
    return {
        "version": "1.0",
        "timestamp": result.timestamp,
        "target": result.target,
        "matches": [asdict(m) for m in result.matches],
    }


def to_human(result: ProbeResult) -> str:
    """Rich-formatted summary."""
    console = Console(record=True, width=120)
    table = Table(title=f"Probe @ {result.target.get('host')}:{result.target.get('port')} (unit {result.target.get('unit_id')})",
                  show_lines=True)
    table.add_column("Match", justify="center")
    table.add_column("Profile")
    table.add_column("Mfg / Model")
    table.add_column("Confidence")
    table.add_column("Notes")
    for m in result.matches:
        if m.match:
            mark = "[green]✓[/green]"
        else:
            mark = "[red]✗[/red]"
        notes = ", ".join(m.failures[:2]) if m.failures else ", ".join(f"{k}={v}" for k, v in list(m.identifiers.items())[:2])
        table.add_row(mark, m.profile_slug, f"{m.manufacturer} / {m.model}", m.confidence, notes)
    console.print(table)
    return console.export_text()


def write_json(result: ProbeResult, path: str) -> None:
    with open(path, "w", encoding="utf-8") as f:
        json.dump(to_json(result), f, indent=2)
```

```python
# packages/probe_cli/src/probe_cli/reference.py
"""--reference-value cross-check: read metric value, compare to operator measurement."""
from __future__ import annotations

from dataclasses import dataclass


@dataclass(frozen=True)
class ReferenceCheck:
    metric: str
    actual: float
    expected: float
    relative_delta: float
    matched: bool


def compare_reference(*, metric: str, actual: float, expected: float,
                      tolerance: float = 0.05) -> ReferenceCheck:
    if abs(expected) < 1e-9:
        delta = abs(actual)
        matched = delta < 1e-3
    else:
        delta = abs(actual - expected) / abs(expected)
        matched = delta <= tolerance
    return ReferenceCheck(metric=metric, actual=actual, expected=expected,
                          relative_delta=delta, matched=matched)
```

```python
# packages/probe_cli/src/probe_cli/commands/identify.py
"""`solamon-probe identify <host>` — identify the device at <host>:<port>."""
from __future__ import annotations

import asyncio
import json
import sys
from pathlib import Path

import typer
from pymodbus.client import AsyncModbusTcpClient

from ..output import to_human, to_json, write_json
from ..runner import run_probe

app = typer.Typer(help="Identify the device at a Modbus TCP host.")

DEFAULT_PROFILES = Path(__file__).resolve().parents[4] / "architecture" / "profiles"
DEFAULT_CATALOG = Path(__file__).resolve().parents[4] / "architecture" / "logical_metrics.yaml"


@app.command()
def identify(
    host: str = typer.Argument(..., help="Modbus TCP host (IP or DNS name)."),
    port: int = typer.Option(502, "--port"),
    unit_id: int = typer.Option(1, "--unit-id"),
    profiles_dir: Path = typer.Option(DEFAULT_PROFILES, "--profiles-dir"),
    catalog_path: Path = typer.Option(DEFAULT_CATALOG, "--catalog"),
    output_json: Path | None = typer.Option(None, "--output-json", help="Write JSON to this path."),
    json_only: bool = typer.Option(False, "--json", help="Emit JSON to stdout (suppress human output)."),
) -> None:
    async def _run():
        client = AsyncModbusTcpClient(host=host, port=port)
        try:
            await client.connect()
            return await run_probe(
                client=client, profiles_dir=profiles_dir, catalog_path=catalog_path,
                unit_id=unit_id, host=host, port=port,
            )
        finally:
            client.close()

    result = asyncio.run(_run())
    if output_json:
        write_json(result, str(output_json))
    if json_only:
        print(json.dumps(to_json(result), indent=2))
    else:
        print(to_human(result))

    matched = [m for m in result.matches if m.match]
    if not matched:
        sys.exit(2)                # nothing matched
```
