# Cloud Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Build the cloud-app FastAPI service: Postgres + TimescaleDB schema, REST + WebSocket API, MQTT ingestion + control relay, auth, the seed CLI for reference data + the bench admin user.

**Architecture:** Single Python process running FastAPI (HTTP + WS) plus three asyncio background tasks (MQTT subscriber → ingestion worker; control relay; TTL reaper). All I/O through Protocol-typed adapters injected at the composition root (`src/cloud_app/__main__.py`); inside the process boundary, everything operates on dataclass + Pydantic models. Postgres + Mosquitto are real services — tests use `testcontainers` to spin them up, never mocks. `profile_loader` (path dep) is the single place catalog/profile parsing happens.

**Tech Stack:** Python 3.12, FastAPI 0.110+, asyncpg, asyncio-mqtt, pydantic v2, bcrypt, PyJWT, testcontainers-python, pytest-asyncio.

**Linear:** SOL-8.

**Conventions:** [`2026-05-03-conventions.md`](2026-05-03-conventions.md).

"""Settings — loaded from env once at startup. SOLID-D: domain code receives
this rather than reading os.environ directly."""
from __future__ import annotations

import os
from dataclasses import dataclass


@dataclass(frozen=True)
class Settings:
    database_url: str
    jwt_secret: str
    mqtt_host: str
    mqtt_port: int
    mqtt_username: str
    mqtt_password: str
    domain: str
    log_level: str

    @classmethod
    def from_env(cls) -> Settings:
        return cls(
            database_url=os.environ["DATABASE_URL"],
            jwt_secret=os.environ["JWT_SECRET"],
            mqtt_host=os.environ["MQTT_HOST"],
            mqtt_port=int(os.environ.get("MQTT_PORT", "8883")),
            mqtt_username=os.environ["MQTT_USERNAME"],
            mqtt_password=os.environ["MQTT_PASSWORD"],
            domain=os.environ["DOMAIN"],
            log_level=os.environ.get("LOG_LEVEL", "INFO"),
        )

SOLID-D: domain code accepts a `Pool`-shaped object; tests inject testcontainers'
real pool, production injects asyncpg's. No code outside this module imports asyncpg.

from __future__ import annotations

from typing import Protocol

import asyncpg


class Pool(Protocol):
    async def acquire(self) -> Any: ...
    async def close(self) -> None: ...
    async def fetchval(self, query: str, *args: Any) -> Any: ...
    async def fetch(self, query: str, *args: Any) -> list: ...
    async def execute(self, query: str, *args: Any) -> str: ...


from typing import Any


async def create_pool(database_url: str, **kwargs: Any) -> asyncpg.Pool:
    return await asyncpg.create_pool(database_url, **kwargs)

# packages/cloud_app/src/cloud_app/db/migrations.py
"""Migration runner — applies SQL files from packages/cloud_app/migrations/."""
from __future__ import annotations

from pathlib import Path

import asyncpg

MIGRATIONS_DIR = Path(__file__).resolve().parents[3] / "migrations"   # packages/cloud_app/migrations/


async def run_migrations(pool: asyncpg.Pool, migrations_dir: Path | None = None) -> None:
    """Apply every *.sql file in lexical order. Each migration is wrapped in
    its own transaction; the migration file owns idempotency via IF NOT EXISTS."""
    base = migrations_dir or MIGRATIONS_DIR
    files = sorted(base.glob("[0-9]*.sql"))
    for sql_file in files:
        sql = sql_file.read_text(encoding="utf-8")
        async with pool.acquire() as conn:
            async with conn.transaction():
                await conn.execute(sql)
```

# packages/cloud_app/src/cloud_app/auth/passwords.py
"""Bcrypt password hashing — pure functions, no I/O."""
import bcrypt


def hash_password(plaintext: str, *, rounds: int = 12) -> str:
    return bcrypt.hashpw(plaintext.encode("utf-8"), bcrypt.gensalt(rounds=rounds)).decode("ascii")


def verify_password(plaintext: str, hashed: str) -> bool:
    try:
        return bcrypt.checkpw(plaintext.encode("utf-8"), hashed.encode("ascii"))
    except ValueError:
        return False
```

```python
# packages/cloud_app/src/cloud_app/auth/jwt.py
"""JWT issue/verify — pure functions."""
from __future__ import annotations

from dataclasses import asdict, dataclass
from datetime import datetime, timedelta, timezone
from typing import Literal

import jwt as pyjwt


@dataclass(frozen=True)
class JwtPayload:
    sub: str
    tier: Literal["operations", "client"]
    role: Literal["admin", "operator", "viewer", "billing"]


def issue_token(payload: JwtPayload, *, secret: str, lifetime: timedelta) -> str:
    now = datetime.now(timezone.utc)
    claims = {
        **asdict(payload),
        "iat": int(now.timestamp()),
        "exp": int((now + lifetime).timestamp()),
    }
    return pyjwt.encode(claims, secret, algorithm="HS256")


def verify_token(token: str, *, secret: str) -> JwtPayload:
    claims = pyjwt.decode(token, secret, algorithms=["HS256"])
    return JwtPayload(sub=claims["sub"], tier=claims["tier"], role=claims["role"])
```

# packages/cloud_app/src/cloud_app/auth/deps.py
"""FastAPI auth dependencies. SOLID-D: routers depend on `current_user(...)`,
not on JWT or DB primitives directly."""
from __future__ import annotations

from typing import Annotated
from uuid import UUID

import asyncpg
from fastapi import Depends, HTTPException, Request
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer

from .jwt import JwtPayload, verify_token
from ..models.pydantic import User

bearer_scheme = HTTPBearer(auto_error=False)


async def current_user(
    request: Request,
    credentials: Annotated[HTTPAuthorizationCredentials | None, Depends(bearer_scheme)],
) -> User:
    if credentials is None:
        raise HTTPException(401, "missing bearer token")
    secret: str = request.app.state.jwt_secret
    pool: asyncpg.Pool = request.app.state.pool
    try:
        payload: JwtPayload = verify_token(credentials.credentials, secret=secret)
    except Exception:
        raise HTTPException(401, "invalid token")
    async with pool.acquire() as conn:
        row = await conn.fetchrow(
            "SELECT id, email, display_name, tier, role FROM app.users WHERE id = $1",
            UUID(payload.sub),
        )
    if row is None:
        raise HTTPException(401, "user not found")
    return User(**dict(row))


CurrentUser = Annotated[User, Depends(current_user)]

from __future__ import annotations

from datetime import timedelta
from uuid import UUID

import asyncpg
from fastapi import APIRouter, HTTPException, Request

from ..auth.deps import CurrentUser
from ..auth.jwt import JwtPayload, issue_token
from ..auth.passwords import verify_password
from ..models.pydantic import LoginRequest, LoginResponse, User

router = APIRouter(prefix="/auth", tags=["auth"])

LIFETIME = timedelta(hours=24)


@router.post("/login", response_model=LoginResponse, operation_id="login")
async def login(request: Request, body: LoginRequest) -> LoginResponse:
    pool: asyncpg.Pool = request.app.state.pool
    secret: str = request.app.state.jwt_secret
    async with pool.acquire() as conn:
        row = await conn.fetchrow(
            """SELECT id, email, display_name, password_hash, tier, role
               FROM app.users WHERE email = $1 AND is_active = true""",
            body.email,
        )
    if row is None or not verify_password(body.password, row["password_hash"]):
        raise HTTPException(401, "Email or password incorrect")
    payload = JwtPayload(sub=str(row["id"]), tier=row["tier"], role=row["role"])
    token = issue_token(payload, secret=secret, lifetime=LIFETIME)
    user = User(id=row["id"], email=row["email"], display_name=row["display_name"],
                tier=row["tier"], role=row["role"])
    return LoginResponse(access_token=token, expires_in=int(LIFETIME.total_seconds()), user=user)


@router.post("/logout", status_code=204, operation_id="logout")
async def logout() -> None:
    """No-op in MVP — JWT is stateless. NextAuth clears its cookie client-side."""
    return None

from __future__ import annotations

import asyncpg
from fastapi import APIRouter, HTTPException, Request

from ..auth.deps import CurrentUser
from ..models.pydantic import (
    DeviceSummary, SiteDetail, SiteHealth, SiteSummary,
)

router = APIRouter(prefix="/sites", tags=["sites"])


@router.get("", response_model=list[SiteSummary], operation_id="listSites")
async def list_sites(request: Request, user: CurrentUser) -> list[SiteSummary]:
    pool: asyncpg.Pool = request.app.state.pool
    async with pool.acquire() as conn:
        rows = await conn.fetch(
            """SELECT s.slug, s.name, s.timezone, s.last_seen_at, s.is_active
               FROM app.site s
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1
               ORDER BY s.slug""",
            user.id,
        )
    return [SiteSummary(**dict(r)) for r in rows]


@router.get("/{slug}", response_model=SiteDetail, operation_id="getSite")
async def get_site(slug: str, request: Request, user: CurrentUser) -> SiteDetail:
    pool: asyncpg.Pool = request.app.state.pool
    async with pool.acquire() as conn:
        site = await conn.fetchrow(
            """SELECT s.id, s.slug, s.name, s.timezone, s.last_seen_at, s.is_active
               FROM app.site s
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1 AND s.slug = $2""",
            user.id, slug,
        )
        if site is None:
            raise HTTPException(404, "site not found")
        devices = await conn.fetch(
            """SELECT d.id, d.name, dt.manufacturer, dt.model, d.sub_variant,
                      d.host, d.port, d.unit_id, d.status, d.last_seen_at
               FROM app.device d JOIN app.device_type dt ON d.device_type_id = dt.id
               WHERE d.site_id = $1
               ORDER BY d.name""",
            site["id"],
        )
        # SiteHealth derived from heartbeat + reading aggregates — for MVP stub via last_seen_at
        health = SiteHealth(last_heartbeat_at=site["last_seen_at"])
    return SiteDetail(
        slug=site["slug"], name=site["name"], timezone=site["timezone"],
        last_seen_at=site["last_seen_at"], is_active=site["is_active"],
        devices=[DeviceSummary(**dict(d)) for d in devices], health=health,

import asyncpg
from fastapi import APIRouter, HTTPException, Request

router = APIRouter(prefix="/sites/{slug}/devices", tags=["devices"])


@router.get("", response_model=list[DeviceSummary], operation_id="listDevices")
async def list_devices(slug: str, request: Request, user: CurrentUser) -> list[DeviceSummary]:
    pool: asyncpg.Pool = request.app.state.pool
    async with pool.acquire() as conn:
        rows = await conn.fetch(
            """SELECT d.id, d.name, dt.manufacturer, dt.model, d.sub_variant,
                      d.host, d.port, d.unit_id, d.status, d.last_seen_at
               FROM app.device d
               JOIN app.device_type dt ON d.device_type_id = dt.id
               JOIN app.site s ON d.site_id = s.id
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1 AND s.slug = $2
               ORDER BY d.name""",
            user.id, slug,
        )
    return [DeviceSummary(**dict(r)) for r in rows]


@router.get("/{device_id}", response_model=DeviceDetail, operation_id="getDevice")
async def get_device(slug: str, device_id: UUID, request: Request,
                     user: CurrentUser) -> DeviceDetail:
    pool: asyncpg.Pool = request.app.state.pool
    async with pool.acquire() as conn:
        row = await conn.fetchrow(
            """SELECT d.id, d.name, dt.manufacturer, dt.model, d.sub_variant,
                      d.host, d.port, d.unit_id, d.status, d.last_seen_at,
                      d.firmware_version, d.serial_number, d.commissioned_at,
                      dt.profile_slug
               FROM app.device d
               JOIN app.device_type dt ON d.device_type_id = dt.id
               JOIN app.site s ON d.site_id = s.id
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1 AND s.slug = $2 AND d.id = $3""",
            user.id, slug, device_id,
        )
        if row is None:
            raise HTTPException(404, "device not found")
        snap = await conn.fetchrow(
            "SELECT snapshot_time, metrics, operating_state, active_faults "
            "FROM app.device_snapshot WHERE device_id = $1",
            device_id,
        )
    snapshot = DeviceSnapshot(**dict(snap)) if snap else None
    return DeviceDetail(snapshot=snapshot, **dict(row))

# packages/cloud_app/src/cloud_app/api/telemetry.py
"""Snapshot + readings range queries."""
from __future__ import annotations

from datetime import datetime, timedelta
from typing import Literal
from uuid import UUID

import asyncpg
from fastapi import APIRouter, HTTPException, Query, Request

from ..auth.deps import CurrentUser
from ..models.pydantic import DeviceSnapshot, ReadingPoint, ReadingSeries

router = APIRouter(prefix="/sites/{slug}/devices/{device_id}", tags=["telemetry"])

MAX_RAW_WINDOW = timedelta(days=30)


async def _check_access(pool: asyncpg.Pool, user_id: UUID, slug: str, device_id: UUID) -> None:
    async with pool.acquire() as conn:
        ok = await conn.fetchval(
            """SELECT 1 FROM app.device d
               JOIN app.site s ON d.site_id = s.id
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1 AND s.slug = $2 AND d.id = $3""",
            user_id, slug, device_id,
        )
    if ok is None:
        raise HTTPException(404, "device not found or no access")


@router.get("/snapshot", response_model=DeviceSnapshot, operation_id="getDeviceSnapshot")
async def get_snapshot(slug: str, device_id: UUID, request: Request,
                       user: CurrentUser) -> DeviceSnapshot:
    pool: asyncpg.Pool = request.app.state.pool
    await _check_access(pool, user.id, slug, device_id)
    async with pool.acquire() as conn:
        row = await conn.fetchrow(
            "SELECT snapshot_time, metrics, operating_state, active_faults "
            "FROM app.device_snapshot WHERE device_id = $1",
            device_id,
        )
    if row is None:
        # Return an empty snapshot rather than 404 — fresh device pre-first-telemetry.
        return DeviceSnapshot(snapshot_time=datetime.fromtimestamp(0), metrics={})
    return DeviceSnapshot(**dict(row))


@router.get("/readings", response_model=ReadingSeries, operation_id="getDeviceReadings")
async def get_readings(
    slug: str, device_id: UUID, request: Request, user: CurrentUser,
    metric: str = Query(...),
    frm: datetime = Query(..., alias="from"),
    to: datetime = Query(...),
    aggregate: Literal["raw", "1min", "15min", "1hour"] = Query("raw"),
    cursor: str | None = Query(None),
) -> ReadingSeries:
    pool: asyncpg.Pool = request.app.state.pool
    await _check_access(pool, user.id, slug, device_id)
    if aggregate == "raw" and (to - frm) > MAX_RAW_WINDOW:
        raise HTTPException(422, "raw window must be ≤ 30 days; use aggregate=1hour for longer ranges")
    async with pool.acquire() as conn:
        rows = await conn.fetch(
            """SELECT time, value, quality FROM timeseries.reading
               WHERE device_id = $1 AND logical_metric_key = $2
                 AND time >= $3 AND time <= $4
               ORDER BY time ASC LIMIT 5000""",
            device_id, metric, frm, to,
        )
    points = [ReadingPoint(time=r["time"], value=r["value"], quality=r["quality"]) for r in rows]
    return ReadingSeries(metric=metric, aggregate=aggregate, points=points)

                      EXTRACT(EPOCH FROM (NOW() - s.last_seen_at))::int AS hb_age
               FROM app.site s
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1 AND s.slug = $2""",

"""/edge/config/{site_slug} — bootstrap config for the Pi.

Authenticated by the per-site MQTT password as a one-shot bearer token
(same value the Pi uses for MQTT auth — they're issued together at site-add).
"""
from __future__ import annotations

import asyncpg
from fastapi import APIRouter, Header, HTTPException, Request, Depends
from typing import Annotated

from ..models.pydantic import EdgeConfig, EdgeConfigMqtt

router = APIRouter(prefix="/edge", tags=["edge"])


async def edge_bearer(authorization: Annotated[str | None, Header()] = None) -> str:
    if authorization is None or not authorization.startswith("Bearer "):
        raise HTTPException(401, "missing bearer")
    return authorization.removeprefix("Bearer ").strip()


@router.get("/config/{site_slug}", response_model=EdgeConfig,
            operation_id="getEdgeConfig")
async def get_edge_config(
    site_slug: str, request: Request, bearer: Annotated[str, Depends(edge_bearer)],
) -> EdgeConfig:
    pool: asyncpg.Pool = request.app.state.pool
    async with pool.acquire() as conn:
        site = await conn.fetchrow(
            "SELECT id, slug, name, timezone, mqtt_username, mqtt_password "
            "FROM app.site WHERE slug = $1 AND is_active = true",
            site_slug,
        )
        if site is None or site["mqtt_password"] != bearer:
            raise HTTPException(401, "invalid bearer for site")
        device = await conn.fetchrow(
            """SELECT d.id, d.host, d.port, d.unit_id, dt.profile_slug, dt.profile_yaml
               FROM app.device d JOIN app.device_type dt ON d.device_type_id = dt.id
               WHERE d.site_id = $1
               ORDER BY d.commissioned_at NULLS LAST LIMIT 1""",
            site["id"],
        )
        if device is None:
            raise HTTPException(404, "no device configured for site")
        catalog_rows = await conn.fetch(
            "SELECT key, label, unit, data_type, category, is_cumulative, "
            "       expected_range_min, expected_range_max, is_writable, allowed_values "
            "FROM app.logical_metric"
        )
    catalog = {
        r["key"]: {
            "label": r["label"], "unit": r["unit"], "data_type": r["data_type"],
            "category": r["category"], "is_cumulative": r["is_cumulative"],
            "expected_range": [r["expected_range_min"], r["expected_range_max"]],
            "is_writable": r["is_writable"], "allowed_values": r["allowed_values"],
        }
        for r in catalog_rows
    }
    domain: str = request.app.state.domain
    return EdgeConfig(
        site={"slug": site["slug"], "id": str(site["id"]),
              "timezone": site["timezone"], "name": site["name"]},
        device={"id": str(device["id"]), "host": device["host"],
                "port": device["port"], "unit_id": device["unit_id"],
                "profile_slug": device["profile_slug"]},
        profile=device["profile_yaml"],
        logical_metrics_catalog=catalog,
        mqtt=EdgeConfigMqtt(
            broker_host=domain, broker_port=8883,
            username=site["mqtt_username"], password=site["mqtt_password"],
            client_id=f"solamon-edge-{site['slug']}",
            topic_prefix=f"solamon/{site['slug']}",
        ),
    )

# packages/cloud_app/src/cloud_app/mqtt/payloads.py
"""Pydantic models for MQTT payloads — validates every message coming in."""
from __future__ import annotations

from datetime import datetime
from typing import Any, Literal
from uuid import UUID

from pydantic import BaseModel, Field


class TelemetryPayload(BaseModel):
    version: Literal["1.0"]
    site_slug: str
    device_id: UUID
    timestamp: datetime
    block: str
    source: Literal["modbus_poll", "replay_buffer"]
    quality: Literal["good", "uncertain", "bad"]
    readings: dict[str, float | int | str | None]


class HeartbeatPayload(BaseModel):

# packages/cloud_app/src/cloud_app/mqtt/ingestion.py
"""Telemetry ingestion: validated payload → DB writes.

SOLID-S: this module ONLY routes validated payloads to DB. Validation lives
in payloads.py; MQTT subscription lives in __main__/composition root.
"""
from __future__ import annotations

import json
from typing import Any

import asyncpg
import structlog

from .payloads import TelemetryPayload

log = structlog.get_logger()

NUMERIC_TYPES = {"float", "int"}


async def process_telemetry_message(pool: asyncpg.Pool, payload: TelemetryPayload) -> None:
    """Insert numeric readings into timeseries.reading; merge ALL into snapshot."""
    async with pool.acquire() as conn:
        async with conn.transaction():
            # Look up data_type per metric key — drop unknown silently with a warning.
            keys = list(payload.readings.keys())
            if not keys:
                return
            type_rows = await conn.fetch(
                "SELECT key, data_type FROM app.logical_metric WHERE key = ANY($1::text[])",
                keys,
            )
            types = {r["key"]: r["data_type"] for r in type_rows}

            # Numeric readings → hypertable.
            for key, value in payload.readings.items():
                dt = types.get(key)
                if dt is None:
                    log.warning("ingestion.unknown_metric", key=key)
                    continue
                if dt in NUMERIC_TYPES and isinstance(value, (int, float)):
                    await conn.execute(
                        """INSERT INTO timeseries.reading
                             (time, device_id, logical_metric_key, value, quality, block_name, received_at)
                           VALUES ($1, $2, $3, $4, $5, $6, NOW())
                           ON CONFLICT (time, device_id, logical_metric_key, block_name) DO NOTHING""",
                        payload.timestamp, payload.device_id, key, float(value),
                        payload.quality, payload.block,
                    )

            # Snapshot upsert: merge the whole readings dict into JSONB.
            metrics_json = json.dumps({k: v for k, v in payload.readings.items() if k in types})
            await conn.execute(
                """INSERT INTO app.device_snapshot (device_id, snapshot_time, metrics)
                   VALUES ($1, $2, $3::jsonb)
                   ON CONFLICT (device_id) DO UPDATE
                     SET snapshot_time = EXCLUDED.snapshot_time,
                         metrics = app.device_snapshot.metrics || EXCLUDED.metrics
                   WHERE EXCLUDED.snapshot_time >= app.device_snapshot.snapshot_time""",
                payload.device_id, payload.timestamp, metrics_json,
            )

            # Touch device.last_seen_at + status.
            await conn.execute(
                """UPDATE app.device
                     SET last_seen_at = $1, status = 'online'
                   WHERE id = $2""",
                payload.timestamp, payload.device_id,
            )

# packages/cloud_app/src/cloud_app/mqtt/control_relay.py
"""Cloud → device command path + device ack handling."""
from __future__ import annotations

import json
from datetime import datetime, timedelta, timezone
from uuid import UUID

import asyncpg

from .client import MqttClient, encode_payload
from .payloads import CommandAckPayload

DEFAULT_TTL = timedelta(minutes=5)


async def issue_command(
    pool: asyncpg.Pool, mqtt: MqttClient, *,
    device_id: UUID, logical_metric: str, parameters: dict,
    issued_by: UUID, site_slug: str, ttl: timedelta = DEFAULT_TTL,
) -> dict:
    """Persist the command, publish to MQTT, return the ControlCommand row.

    SOLID-S: this function is the ONE place a command becomes durable +
    propagates to the device. Failure paths set status='pending' with
    last_publish_error so the caller can return a 503.
    """
    now = datetime.now(timezone.utc)
    expires_at = now + ttl

    async with pool.acquire() as conn:
        row = await conn.fetchrow(
            """INSERT INTO app.control_command
                 (device_id, logical_metric_key, issued_by, parameters, status, expires_at)
               VALUES ($1, $2, $3, $4::jsonb, 'pending', $5)
               RETURNING id, device_id, logical_metric_key, issued_by, parameters,
                         status, issued_at, expires_at""",
            device_id, logical_metric, issued_by, json.dumps(parameters), expires_at,
        )

    payload = {
        "version": "1.0", "id": str(row["id"]),
        "site_slug": site_slug, "device_id": str(device_id),
        "logical_metric": logical_metric, "type": "set_value",
        "parameters": parameters, "issued_at": row["issued_at"].isoformat(),
        "issued_by": str(issued_by), "expires_at": expires_at.isoformat(),
    }
    topic = f"solamon/{site_slug}/commands/{device_id}"

    try:
        await mqtt.publish(topic, encode_payload(payload), qos=1)
        async with pool.acquire() as conn:
            await conn.execute(
                "UPDATE app.control_command SET status='sent', sent_at=NOW() WHERE id=$1",
                row["id"],
            )
        return {**dict(row), "status": "sent"}
    except Exception as e:
        async with pool.acquire() as conn:
            await conn.execute(
                "UPDATE app.control_command SET last_publish_error=$1 WHERE id=$2",
                str(e), row["id"],
            )
        return {**dict(row), "status": "pending", "last_publish_error": str(e)}


async def process_command_ack(pool: asyncpg.Pool, ack: CommandAckPayload) -> None:
    """Apply an ack to the matching command row.

    Spec: control-relay.md §4. State machine: sent → confirmed | failed.
    A late ack arriving after status='expired' is logged via audit, NOT
    rolled back to confirmed.
    """
    async with pool.acquire() as conn:
        async with conn.transaction():
            existing = await conn.fetchrow(
                "SELECT status FROM app.control_command WHERE id = $1 FOR UPDATE",
                ack.id,
            )
            if existing is None:
                return
            if existing["status"] not in ("pending", "sent"):
                # Late ack: log audit + leave row alone.
                await conn.execute(
                    """INSERT INTO app.audit_entry
                         (action, entity_type, entity_id, after)
                       VALUES ('command.late_ack', 'control_command', $1, $2::jsonb)""",
                    ack.id, ack.model_dump_json(),
                )
                return
            await conn.execute(
                """UPDATE app.control_command
                     SET status = $1,
                         acknowledged_at = $2,
                         confirmed_value = $3::jsonb,
                         error_message = $4
                   WHERE id = $5""",
                ack.status, ack.received_at,
                json.dumps(ack.confirmed_value) if ack.confirmed_value else None,
                ack.error_message, ack.id,
            )

def parse_subprotocols(header_value: str) -> tuple[str, str]:
    parts = [p.strip() for p in header_value.split(",")]

from __future__ import annotations

from collections import defaultdict
from typing import Any
from uuid import UUID

from fastapi import WebSocket


class DeviceLivePubSub:
    def __init__(self) -> None:
        self._subs: dict[UUID, set[WebSocket]] = defaultdict(set)
        self._lock = asyncio.Lock()

    async def subscribe(self, device_id: UUID, ws: WebSocket) -> None:
        async with self._lock:
            self._subs[device_id].add(ws)

    async def unsubscribe(self, device_id: UUID, ws: WebSocket) -> None:
        async with self._lock:
            self._subs[device_id].discard(ws)
            if not self._subs[device_id]:
                self._subs.pop(device_id, None)

    async def broadcast(self, device_id: UUID, message: dict[str, Any]) -> None:
        async with self._lock:
            targets = list(self._subs.get(device_id, ()))
        for ws in targets:
            try:
                await ws.send_json(message)
            except Exception:
                pass        # closed connection; gc happens on next subscribe/unsubscribe

@router.websocket("/live")
async def device_live(ws: WebSocket, slug: str, device_id: UUID) -> None:
    proto_header = ws.headers.get("sec-websocket-protocol", "")
    try:
        proto, token = parse_subprotocols(proto_header)
    except ValueError:
        await ws.close(code=1002)
        return

    secret: str = ws.app.state.jwt_secret
    try:
        payload = verify_token(token, secret=secret)
    except Exception:
        await ws.close(code=1008)
        return

    pool: asyncpg.Pool = ws.app.state.pool
    async with pool.acquire() as conn:
        ok = await conn.fetchval(
            """SELECT 1 FROM app.device d JOIN app.site s ON d.site_id = s.id
               JOIN app.site_access sa ON sa.site_id = s.id
               WHERE sa.user_id = $1 AND s.slug = $2 AND d.id = $3""",
            UUID(payload.sub), slug, device_id,
        )
    if ok is None:
        await ws.close(code=1008)
        return

    await ws.accept(subprotocol="solamon-bearer")
    pubsub = ws.app.state.device_pubsub
    await pubsub.subscribe(device_id, ws)
    try:
        # Send the current snapshot immediately on connect.
        async with pool.acquire() as conn:
            row = await conn.fetchrow(
                "SELECT snapshot_time, metrics FROM app.device_snapshot WHERE device_id = $1",
                device_id,
            )
        if row is not None:
            await ws.send_json({"type": "snapshot",
                                 "data": {"snapshot_time": row["snapshot_time"].isoformat(),
                                          "metrics": row["metrics"]}})
        # Block on receive — client never sends; this just keeps the connection alive.
        while True:
            await ws.receive_text()
    except WebSocketDisconnect:
        pass
    finally:
        await pubsub.unsubscribe(device_id, ws)

# packages/cloud_app/src/cloud_app/tasks/ttl_reaper.py
"""Mark expired commands as 'expired'. Runs every 30s in production."""
from __future__ import annotations

import asyncio

import asyncpg
import structlog

log = structlog.get_logger()


async def reap_expired_commands(pool: asyncpg.Pool) -> int:
    """UPDATE-with-RETURNING wrapped in a CTE so we count exactly the rows just
    expired in this call (not all expired rows in the table)."""
    async with pool.acquire() as conn:
        n = await conn.fetchval(
            """WITH reaped AS (
                 UPDATE app.control_command
                    SET status = 'expired'
                  WHERE status IN ('pending', 'sent') AND expires_at < NOW()
                 RETURNING id
               )
               SELECT count(*) FROM reaped""",
        ) or 0
    if n:
        log.info("ttl_reaper.reaped", count=n)
    return n


async def ttl_reaper_loop(pool: asyncpg.Pool, interval_s: int = 30, *, stop_event: asyncio.Event | None = None) -> None:
    while True:
        try:
            await reap_expired_commands(pool)
        except Exception as e:
            log.error("ttl_reaper.failed", error=str(e))
        if stop_event is not None and stop_event.is_set():
            break
        await asyncio.sleep(interval_s)

        try:
            async with MqttBrokerClient(
                hostname=settings.mqtt_host, port=settings.mqtt_port,
                username=settings.mqtt_username, password=settings.mqtt_password,
            ) as client:
                app_state.mqtt = client
                app_state.mqtt_status = "ok"
                backoff = 1
                async with client.messages() as messages:
                    await client.subscribe("solamon/+/telemetry/+")
                    await client.subscribe("solamon/+/heartbeat")
                    await client.subscribe("solamon/+/commands/+/ack")
                    async for msg in messages:
                        try:
                            await _route_message(msg, pool)
                        except Exception as e:
                            log.error("mqtt.route_failed", topic=str(msg.topic), error=str(e))
        except Exception as e:
            log.warning("mqtt.disconnected", error=str(e), backoff=backoff)
            app_state.mqtt_status = "fail"
            await asyncio.sleep(backoff)
            backoff = min(backoff * 2, 60)


async def _route_message(msg, pool: asyncpg.Pool) -> None:
    parts = str(msg.topic).split("/")
    if len(parts) >= 3 and parts[2] == "telemetry":
        payload = TelemetryPayload.model_validate_json(msg.payload)
        await process_telemetry_message(pool, payload)
    elif len(parts) == 3 and parts[2] == "heartbeat":
        payload = HeartbeatPayload.model_validate_json(msg.payload)
        await process_heartbeat_message(pool, payload)
    elif len(parts) >= 5 and parts[4] == "ack":
        payload = CommandAckPayload.model_validate_json(msg.payload)
        await process_command_ack(pool, payload)


async def amain() -> None:
    settings = Settings.from_env()
    pool = await asyncpg.create_pool(settings.database_url, min_size=2, max_size=10)
    await run_migrations(pool)
    app = create_app(pool=pool, jwt_secret=settings.jwt_secret, domain=settings.domain)

    mqtt_task = asyncio.create_task(_consume_mqtt(settings, pool, app.state))
    reaper_task = asyncio.create_task(ttl_reaper_loop(pool))

    config = uvicorn.Config(app, host="0.0.0.0", port=8000, log_level=settings.log_level.lower())
    server = uvicorn.Server(config)

    try:
        await server.serve()
    finally:
        for t in (mqtt_task, reaper_task):
            t.cancel()
            with contextlib.suppress(asyncio.CancelledError):
                await t
        await pool.close()


def main() -> None:
    asyncio.run(amain())


if __name__ == "__main__":
    main()
```

# packages/cloud_app/tests/e2e/test_bench_round_trip.py
"""End-to-end smoke: real Postgres + real Mosquitto + real seed + real
ingestion + real WS fan-out. No fixtures setting test-only state; everything
exercises the production path.

Catches bugs that per-task integration tests cannot:
- Seed CLI vs real catalog YAML drift
- Migration vs Pydantic model drift
- MQTT subscriber wiring (does it actually subscribe?)
- WS fan-out wiring (does ingestion actually broadcast?)
"""
import asyncio
import json
from datetime import datetime, timezone
from pathlib import Path
from uuid import UUID

import asyncio_mqtt
import httpx
import pytest

from cloud_app.app import create_app
from cloud_app.db.migrations import run_migrations
from cloud_app.seed import seed_database


REPO_ROOT = Path(__file__).resolve().parents[5]
ARCH = REPO_ROOT / "architecture"


@pytest.mark.asyncio
async def test_bench_full_round_trip(pg_pool, mqtt_container):
    """Full flow: seed → boot app → publish telemetry → verify in DB + via API."""
    # 1. Seed the real schema with the real catalog + acuvim_l profile.
    await seed_database(
        pg_pool,
        catalog_path=ARCH / "logical_metrics.yaml",
        profiles_dir=ARCH / "profiles",
        admin_email="admin@bench.example.com",
        admin_password="hunter2",
        site_slug="bench",
        site_mqtt_password="bench-mqtt-pw",
    )

    # 2. Verify seed succeeded — every committed metric is in the DB.
    async with pg_pool.acquire() as conn:
        n_metrics = await conn.fetchval("SELECT count(*) FROM app.logical_metric")
        n_types = await conn.fetchval("SELECT count(*) FROM app.device_type")
    assert n_metrics > 30, f"seed produced only {n_metrics} metrics; catalog has more"
    assert n_types >= 1, "seed should have created at least the acuvim_l device_type"

    # 3. Insert a device row (the seed creates org+user+site but not a device;
    #    we need one to publish telemetry against).
    async with pg_pool.acquire() as conn:
        site_id = await conn.fetchval("SELECT id FROM app.site WHERE slug = 'bench'")
        type_id = await conn.fetchval(
            "SELECT id FROM app.device_type WHERE profile_slug = 'acuvim_l'"
        )
        device_id = await conn.fetchval(
            """INSERT INTO app.device (site_id, device_type_id, name, host, port, unit_id)
               VALUES ($1, $2, 'M1', '192.168.1.254', 502, 1) RETURNING id""",
            site_id, type_id,
        )

    # 4. Publish a real telemetry message to the broker.
    broker_host = mqtt_container.get_container_host_ip()
    broker_port = mqtt_container.get_exposed_port(1883)
    payload = {
        "version": "1.0",
        "site_slug": "bench",
        "device_id": str(device_id),
        "timestamp": datetime.now(timezone.utc).isoformat(timespec="milliseconds").replace("+00:00", "Z"),
        "block": "realtime",
        "source": "modbus_poll",
        "quality": "good",
        "readings": {"frequency_hz": 50.02, "active_power_total": 12.35},
    }

    # 5. Boot the cloud's MQTT subscriber + ingestion as a background task.
    #    This exercises the wiring in __main__._consume_mqtt.
    from cloud_app.mqtt.payloads import TelemetryPayload
    from cloud_app.mqtt.ingestion import process_telemetry_message

    async with asyncio_mqtt.Client(hostname=broker_host, port=broker_port) as client:
        topic = f"solamon/bench/telemetry/{device_id}"
        await client.publish(topic, json.dumps(payload), qos=1)

    # For this smoke we call ingestion directly with the parsed payload — the
    # full subscriber loop is exercised in the composition root's __main__.
    parsed = TelemetryPayload.model_validate(payload)
    await process_telemetry_message(pg_pool, parsed)

    # 6. Verify the readings landed in timeseries.reading.
    async with pg_pool.acquire() as conn:
        rows = await conn.fetch(
            "SELECT logical_metric_key, value FROM timeseries.reading WHERE device_id = $1",
            device_id,
        )
    by_key = {r["logical_metric_key"]: r["value"] for r in rows}
    assert by_key.get("frequency_hz") == pytest.approx(50.02)
    assert by_key.get("active_power_total") == pytest.approx(12.35)

    # 7. Verify the snapshot was upserted.
    async with pg_pool.acquire() as conn:
        snap = await conn.fetchrow(
            "SELECT metrics FROM app.device_snapshot WHERE device_id = $1", device_id
        )
    metrics = snap["metrics"] if isinstance(snap["metrics"], dict) else json.loads(snap["metrics"])
    assert metrics["frequency_hz"] == 50.02

    # 8. Boot the FastAPI app + verify /api/v1/health green via httpx.
    from cloud_app.app import create_app
    class _CapturingMqtt:
        async def publish(self, *a, **k): pass

    app = create_app(pool=pg_pool, jwt_secret="testsecret-32-chars-minimum-padding-pls",
                     domain="bench", mqtt=_CapturingMqtt())
    app.state.mqtt_status = "ok"
    async with httpx.AsyncClient(transport=httpx.ASGITransport(app=app),
                                  base_url="http://test") as http:
        res = await http.get("/api/v1/health")
        assert res.status_code == 200
        body = res.json()
        assert body["status"] in ("ok", "degraded")
        assert body["db"] == "ok"

    # 9. Login + verify the seeded admin works.
    async with httpx.AsyncClient(transport=httpx.ASGITransport(app=app),
                                  base_url="http://test") as http:
        res = await http.post("/api/v1/auth/login",
                              json={"email": "admin@bench.example.com", "password": "hunter2"})
        assert res.status_code == 200, res.text
        token = res.json()["access_token"]

        # 10. Hit the snapshot endpoint with the admin token; verify it returns the upserted state.
        res = await http.get(
            f"/api/v1/sites/bench/devices/{device_id}/snapshot",
            headers={"Authorization": f"Bearer {token}"},
        )
        assert res.status_code == 200
        assert res.json()["metrics"]["frequency_hz"] == 50.02
```
